TSW Test Suite

Core

New SoundWorld

            var context = new webkitAudioContext(),
                tsw = new SoundWorld(context);
            // tsw is your new Sound World!

Load file

            tsw.load({
                guitar: 'audio/guitar.ogg',
                bass: 'audio/bass.wav'
            }, function (buffers) {
                // buffers are ready to play 
            });

Play file

            tsw.load({
                guitar: 'audio/guitar.ogg'
            }, function (buffers) {
                var guitar = buffers.guitar;

                tsw.connect(guitar, tsw.speakers);
                tsw.play(guitar);
            });

Connect nodes

            var sineWave = tsw.createSineWave(),
                distortion = tsw.createDistortion();

            tsw.connect(sineWave, distortion, tsw.speakers);

Reverse

            var backwardGuitar = tsw.reverse(guitar);

            tsw.play(backwardGuitar);

Mute

            tsw.mute();
